ingestion/chunkProcessor.ts

Divide documentos grandes.

export function splitIntoChunks(text: string) {

  const chunkSize = 500

  const chunks = []

  for (let i = 0; i < text.length; i += chunkSize) {

    chunks.push(text.substring(i, i + chunkSize))

  }

  return chunks

}
embeddings/embeddingService.ts

Gera embeddings.

export async function generateEmbedding(text: string) {

  const response = await fetch("https://api.deepseek.com/embeddings", {

    method: "POST",

    headers: {
      "Content-Type": "application/json",
      Authorization: `Bearer ${process.env.DEEPSEEK_API_KEY}`
    },

    body: JSON.stringify({

      model: "deepseek-embedding",
      input: text

    })

  })

  const data = await response.json()

  return data.data[0].embedding

}

generation/ragPipeline.ts

Pipeline RAG completo.

import { searchKnowledge } from "../retrieval/vectorSearch"

export async function generateRAGResponse(question: string) {

  const docs = await searchKnowledge(question)

  const context = docs.join("\n")

  const prompt = `
Use o seguinte conhecimento veterinário para responder:

${context}

Pergunta:

${question}
`

  const response = await fetch("https://api.deepseek.com/chat/completions", {

    method: "POST",

    headers: {
      "Content-Type": "application/json",
      Authorization: `Bearer ${process.env.DEEPSEEK_API_KEY}`
    },

    body: JSON.stringify({

      model: "deepseek-chat",

      messages: [

        {
          role: "system",
          content: "Você responde baseado em literatura veterinária."
        },

        {
          role: "user",
          content: prompt
        }

      ]

    })

  })

  const data = await response.json()

  return data.choices[0].message.content

}

AI RAG Engine — AgendaVet

Sistema de Retrieval Augmented Generation para fornecer respostas baseadas em literatura veterinária.

Objetivo

Permitir que a IA responda usando conhecimento científico em vez de apenas conhecimento do modelo.
