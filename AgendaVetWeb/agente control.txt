FRONTEND (Dashboard)
frontend/src/api.ts
export async function getAgents() {

  const res = await fetch("http://localhost:4000/agents")

  return res.json()

}

export async function getMetrics() {

  const res = await fetch("http://localhost:4000/metrics")

  return res.json()

}

export async function deployPrompt(prompt: string) {

  const res = await fetch("http://localhost:4000/prompts/deploy", {
    method: "POST",
    headers: {
      "Content-Type": "application/json"
    },
    body: JSON.stringify({ prompt })
  })

  return res.json()

}
Dashboard principal
frontend/src/pages/Dashboard.tsx
import { useEffect, useState } from "react"
import { getMetrics } from "../api"

export default function Dashboard() {

  const [metrics, setMetrics] = useState<any>()

  useEffect(() => {

    getMetrics().then(setMetrics)

  }, [])

  if (!metrics) return <div>Carregando...</div>

  return (

    <div>

      <h1>AI Control Tower</h1>

      <p>Agentes totais: {metrics.totalAgents}</p>
      <p>Agentes ativos: {metrics.activeAgents}</p>
      <p>Tokens hoje: {metrics.tokensToday}</p>
      <p>Custo estimado: ${metrics.estimatedCost}</p>

    </div>

  )

}
Lista de agentes
frontend/src/pages/Agents.tsx
import { useEffect, useState } from "react"
import { getAgents } from "../api"

export default function Agents() {

  const [agents, setAgents] = useState<any[]>([])

  useEffect(() => {

    getAgents().then(setAgents)

  }, [])

  return (

    <div>

      <h2>Agentes ativos</h2>

      {agents.map(agent => (

        <div key={agent.id}>

          <p>{agent.vet}</p>
          <p>Status: {agent.status}</p>
          <p>Tokens hoje: {agent.tokensToday}</p>

        </div>

      ))}

    </div>

  )

}
Deploy de prompts
frontend/src/components/PromptDeploy.tsx
import { useState } from "react"
import { deployPrompt } from "../api"

export default function PromptDeploy() {

  const [prompt, setPrompt] = useState("")

  async function handleDeploy() {

    await deployPrompt(prompt)

    alert("Prompt enviado")

  }

  return (

    <div>

      <h3>Deploy de Prompt</h3>

      <textarea
        value={prompt}
        onChange={(e) => setPrompt(e.target.value)}
      />

      <button onClick={handleDeploy}>
        Enviar para todos agentes
      </button>

    </div>

  )

}

AI Control Tower - AgendaVet

Painel central de monitoramento e controle dos agentes de IA do sistema AgendaVet.

Funcionalidades

Monitoramento de agentes ativos

Estatísticas de uso de tokens

Estimativa de custo de IA

Deploy de prompts

Visualização de clínicas conectadas

↓

Agent Control System

↓

Agentes IA por veterinário

Endpoints

Lista agentes ativos

GET /metrics
Estatísticas globais

POST /prompts/deploy
Atualiza prompts para todos agentes
